# Algoritma Pengindeksan Corpus Semantik Al-Quran
ALGORITHM IndexingPhase
    INPUT: quran_verses_table, word_embedding_model
    OUTPUT: SQLite verse_vectors_table

    FOR EACH verse IN quran_verses_table:
        // 1. Pra-pemrosesan teks
        text = verse.translation
        text_lowercase = lowercase(text)
        text_cleaned = remove_punctuation(text_lowercase)
        tokens = tokenize(text_cleaned)
        filtered_tokens = remove_stopwords(tokens)

        // 2. Pembuatan representasi vektor
        word_vectors = []
        FOR EACH token IN filtered_tokens:
            IF token IN word_embedding_model:
                vector = word_embedding_model[token]
                word_vectors.append(vector)
            ELSE:
                // Kata di luar kamus (OOV) diabaikan atau ditangani model
                CONTINUE
        
        IF word_vectors NOT EMPTY:
            // Rata-rata pooling & L2 Normalization
            average_vector = mean(word_vectors, axis=0)
            final_vector = l2_normalize(average_vector)
        ELSE:
            final_vector = zero_vector(200)

        // 3. Simpan Vektor ke SQLite DB
        sqlite_save_blob(verse.id, final_vector)
    END FOR
END ALGORITHM
indexing_pipeline.py
# Python implementation code of Semantic Search Indexing Phase

import numpy as np
from backend.preprocessing import remove_punctuation, remove_stopwords, tokenize

def preprocess_text(text):
    lowercased = text.lower()
    no_punct = remove_punctuation(lowercased)
    tokens = tokenize(no_punct)
    filtered = remove_stopwords(tokens)
    return filtered

def generate_verse_vector(model, filtered_tokens):
    word_vectors = []
    for token in filtered_tokens:
        try:
            if token in model.wv:
                word_vectors.append(model.wv[token])
        except Exception:
            continue
            
    if not word_vectors:
        return np.zeros(200)
        
    # Mean pooling
    mean_vec = np.mean(word_vectors, axis=0)
    # L2 normalization
    norm = np.linalg.norm(mean_vec)
    if norm == 0:
        return mean_vec
    return mean_vec / norm

def save_to_database(conn, verse_id, model_type, vector):
    # Serialize float vector to sqlite BLOB
    vector_blob = vector.astype(np.float32).tobytes()
    cursor = conn.cursor()
    cursor.execute(
        "INSERT INTO verse_vectors (verse_id, model_type, vector_blob) VALUES (?, ?, ?)",
        (verse_id, model_type, sqlite3.Binary(vector_blob))
    )
    conn.commit()
Metodologi & Parameter Ekspor Data
1. Dasar Teori & Formulasi Matematika

Dalam metode Information Retrieval berbasis semantik, fasa pengindeksan mentransformasikan teks terjemahan ayat Al-Quran menjadi representasi vektor numerik. Proses ini melalui tiga tahapan matematika:

  1. Pra-pemrosesan Teks: Pembersihan teks mentah terjemahan ($d$) menjadi sekumpulan token bersih ($T$):
    T = remove_stopwords(tokenize(remove_punctuation(lowercase(d))))
  2. Mean Pooling (Rata-rata Vektor Kata): Pembentukan vektor representasi dokumen ($\mathbf{v}_{mean}$) dengan merata-ratakan seluruh vektor kata anggota token bersih yang terdaftar dalam model embedding:
    \(\mathbf{v}_{mean} = \frac{1}{N} \sum_{i=1}^{N} \mathbf{w}_i\)
    Di mana $N$ adalah jumlah kata yang ditemukan dalam kosakata model, dan $\mathbf{w}_i$ adalah vektor embedding dari kata ke-$i$.
  3. L2 Normalization (Normalisasi Vektor): Menormalkan panjang vektor dokumen menjadi 1.0 (unit vector) agar pencarian kemiripan kosinus (Cosine Similarity) di fasa pencarian konsisten:
    \(\mathbf{v}_{final} = \frac{\mathbf{v}_{mean}}{\|\mathbf{v}_{mean}\|_2}\)
    Di mana $\|\mathbf{v}_{mean}\|_2$ adalah magnitudo L2-norm yang dihitung dengan rumus:
    \(\|\mathbf{v}_{mean}\|_2 = \sqrt{\sum_{j=1}^{D} v_j^2}\)
    dengan $D = 200$ (dimensi ruang vektor model Word2Vec/FastText/GloVe).

-- Skema tabel untuk menyimpan terjemahan ayat
CREATE TABLE quran_verses (
    id INTEGER PRIMARY KEY,
    surah_id INTEGER,
    verse_number INTEGER,
    verse_text TEXT,
    verse_translation TEXT
);

-- Skema tabel untuk menyimpan vektor embedding hasil pengindeksan
CREATE TABLE verse_vectors (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    verse_id TEXT UNIQUE, -- Format: "surah_number:verse_number"
    model_type TEXT,       -- 'word2vec', 'fasttext', 'glove'
    vector_blob BLOB       -- Representasi biner array float32 (800 bytes)
);
